import { Card, CardGrid } from '@astrojs/starlight/components';

logfwd replaces complex log pipelines with a single binary that reads, transforms, and forwards logs. Write standard SQL to control exactly what gets shipped — no vendor DSL, no agents to coordinate, no runtime dependencies.

<CardGrid>
  <Card title="SQL Transforms" icon="pencil">
    Filter, reshape, and enrich every batch with standard SQL. `SELECT`, `WHERE`, `GROUP BY`, JOINs, and custom UDFs like `grok()`, `regexp_extract()`, and `geo_lookup()`.
  </Card>
  <Card title="SIMD-Powered Scanner" icon="rocket">
    Parse JSON and CRI at ~2.8M lines/sec per core. Zero-copy Arrow RecordBatches via platform-native SIMD (AVX2, SSE2, NEON).
  </Card>
  <Card title="Single Binary" icon="seti:rust">
    One static binary, no runtime dependencies. Download, configure, run. Ships as a container image for Kubernetes DaemonSets.
  </Card>
  <Card title="OTLP Native" icon="star">
    First-class OpenTelemetry support. Protobuf encoding with zstd compression. Works with any OTLP-compatible collector.
  </Card>
</CardGrid>
